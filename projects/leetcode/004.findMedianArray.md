```
var findMedianSortedArrays = function(nums1, nums2) {
  const n = nums1.length + nums2.length
  let mid1, mid2
  let arr = nums1.concat(nums2)
  arr.sort((a, b) => {
    return a - b
  })
  if (n % 2 == 1) {
    mid1 = arr[(n - 1) / 2]
    return mid1 / 1.0
  } else {
    mid2 = (arr[n / 2] + arr[n / 2 - 1]) / 2.0
    return mid2
  }
};
```
